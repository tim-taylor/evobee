300 // Collect pollen from flower if available, and add to our store
301 // If any pollen is added, the contents of the store is shuffled so that
302 // subsequent operations that act of randomly chosen grains from the store
303 // can simply and efficiently pick off grains from the end of the store.
304 // If the store's maxium capacity is exceeded, some random grains are
305 // removed to bring it down to the maximum allowed size.
306 //
307 void Pollinator::collectPollenFromAnther(Flower* pFlower)
308 {
309  // transfer anther pollen from flower to our store
310  int num = pFlower->transferAntherPollenToPollinator(m_PollenStore);
311 
312  // if some new grains have been added, shuffle the whole store now
313  // so that subsequent removals of pollen happen in a random order
314  if (num > 0)
315  {
316  std::shuffle(m_PollenStore.begin(), m_PollenStore.end(), EvoBeeModel::m_sRngEngine);
317  }
318 
319  // finally, if the number of grains in our store now exceeds the
320  // maximum capacity, delete some. As we've just shuffled the store,
321  // we can just remove the required number from the end of the store
322  int xs = m_PollenStore.size() - m_iMaxPollenCapacity;
323  if (xs > 0)
324  {
325  m_PollenStore.erase(m_PollenStore.end()-xs, m_PollenStore.end());
326  }
327 }
