36  * and
37  * https://en.wikipedia.org/wiki/L%C3%A9vy_distribution
38  *
39  * @param max The maximum possible return value
40  * @param scale Determines positions of the peak of the Levy PDF (must be > 0.0)
41  *
42  */

44  {
45  float x, y, pdf, ymax;
46 
47  if (scale <= 0.0) {
48  throw std::runtime_error("In randomLevy(): scale parameter must be greate than 0.0!");
49  }
50

52  // maximum of the Levy PDF at difference values of the scale parameter
53  if (scale < 1.0) {
54  ymax = 1.0;
55  } else if (scale < 2.0) {
56  ymax = 0.5;
57  } else {

59  }

63  y = EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) * ymax;
64  pdf = std::sqrt(scale/EvoBee::TWOPI) * std::exp(-scale/(2.0*x)) / (std::pow(x,1.5));
65  } while (y > pdf);

70  /*
71  * Return a random variate from a standard Cauchy distribution (with scale=1.0
72  * and shift=0.0).
73  *
74  * A Cauchy distribution is a stable distribution with a PDF f(x) is approximately
75  * proportional to x^-2. To be exact, the PDF of the standard Cauchy distribution
76  * is f(x)=1/(PI.(1+x^2)). It is therefore a suitable representation of the
77  * Levy distribution with an inverse-square law tail that Reynolds et al (2007)
78  * found to be a good fit to observations of real honeybee flight paths.
79  *
80  * To adjust the distribution to be suitable for use in our simulations, this method
81  * always returns a positive value (by taking the absolute value of the random
82  * variate returned from the Cauchy distribution). Furthermore, we define
83  * a minimum and maximum value that this method can return. Values below the minimum
84  * or above the maximum are replaced with the minimum allowed value.
85  *
86  * The maximum has a default value of 20.0 perceptual distance units (pdus), which
87  * corresponds to 14m. A honeybee has a typical maximum speed of ~25kmh = 7m/s.
88  * Hence 20pdus is the maximum a honeybee might be able to fly in about 2 seconds.
89  *
90  * References:
91  * - Andrew M. Reynolds et al. "Displaced Honey Bees Peform Optimal Scale-Free Search
92  * Flights", Ecology, 88(8), 2007, pp. 1955-1961.
93  * - Oliver C. Ibe, "Elements of Random Walk and Diffusion Processes" (Chapter 8,
94  * "Levy Walk"), John Wiley & Sons, 2013.
95  *
96  * @param min The minimum possible return value
97  * @param max The maximum possible return value
98  *
99  */
100  float randomCauchy(float min /*=0.5*/, float max /*=20.0*/)
101  {
102  float length = std::abs(EvoBeeModel::m_sCauchyProbDistrib(EvoBeeModel::m_sRngEngine));
103 
104  if (length < min || length > max) {
105  length = min;
106  }
107 
108  return length;
109  }
