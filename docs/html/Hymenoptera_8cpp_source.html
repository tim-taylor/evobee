212 // When using arbitrary dominant wavelengths, we pick a target wavelength by looking at the set of available
213 // dominant wavelengths (as defined in the PlantTypeConfigs), find the giurfa preference for each of those, normalise
214 // the list of found preferences, and use that as the basis from which to pick a dominant wavelength.
215 void Hymenoptera::initialiseInnateTargetArbitrary()
216 {
217  // This method makes use of the following static map that is populated the first time we call the method.
218  // It maps a dominant wavelength to a tuple comprising the following elements:
219  // 0. (int) Number of PlantTypeConfigs that match this dominant wavelength
220  // 1. (float) The Giurfa preference for this dominant wavelength
221  // 2. (float) The raw (unnormalised) probability of picking this wavelength (PlambdaRaw)
222  // 3. (float) The normalised probability of picking this wavelength (PlambdaNorm)
223  // 4. (float) The cumulative sum of PlambdaNorms for all wavelengths up to and including this one
224  // 5. (vector) A list of pointers to all the PlantTypeConfigs that have this wavelength
225  static std::map<Wavelength, std::tuple<int, float, float, float, float, std::vector<const PlantTypeConfig*>>> prefData;
226  static bool cumulativeInnatePrefsInitialised = false;
227 
228  if (!cumulativeInnatePrefsInitialised) {
229  auto& ptcs = ModelParams::getPlantTypeConfigs();
230  for (auto& ptc : ptcs) {
231  Wavelength lambda = ptc.getDominantWavelength();
232  auto it = prefData.find(lambda);
233  if (it == prefData.end()) {
234  std::vector<const PlantTypeConfig*> ptcvec {&ptc};
235  float gpref = getGiurfaPref(lambda);
236  prefData.insert(std::make_pair(lambda, std::make_tuple(1, gpref, 0.0f, 0.0f, 0.0f, ptcvec)));
237  }
238  else {
239  std::get<0>(it->second)++; // increment count of number of PTCs with this wavelength
240  std::get<5>(it->second).push_back(&ptc); // add a pointer to this PTC to the vector of pointers
241  }
242  }
243 
244  float numPTCs = (float)(ptcs.size());
245  float sumPlambdaRaw = 0.0;
246 
247  for (auto& entry : prefData) {
248  // calculate and set PlambdaRaw for this wavelength
249  float PlambdaRaw = (((float)std::get<0>(entry.second)) * std::get<1>(entry.second)) / numPTCs;
250  std::get<2>(entry.second) = PlambdaRaw;
251  // and update running count of sum of all PlambdaRaws
252  sumPlambdaRaw += PlambdaRaw;
253  }
254 
255  float cumPlambdaNorm = 0.0;
256  for (auto& entry : prefData) {
257  // calculate and set PlambdaNorm for this wavelength
258  float PlambdaNorm = std::get<2>(entry.second) / sumPlambdaRaw;
259  std::get<3>(entry.second) = PlambdaNorm;
260  // and record the cumulative sum of all PlambdaNorms up to this point
261  cumPlambdaNorm += PlambdaNorm;
262  std::get<4>(entry.second) = cumPlambdaNorm;
263  }
264 
265  cumulativeInnatePrefsInitialised = true;
266  }
267 
268 
269  // to assign a target we use roulette wheel selection, picking a number at random between [0,1) and selecting
270  // the first wavelength whose entry shows a cumulative probability greater than that random number
271  float selection = EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine);
272 
273  // std::map is a sorted container, so we can be sure when we are traversing its members that we are
274  // going sequentially from lowest to highest wavelength.
275  // The second test below involving prefData.size() checks whether this is the last entry in
276  // the container. We do this extra check just to avoid any floating point problems when doing
277  // comparisons, to ensure that the last entry is definitely selected if we get to that point.
278  std::size_t item = 1;
279  for (auto& entry : prefData) {
280  if ((std::get<4>(entry.second) >= selection) || (item++ == prefData.size())) {
281  std::uniform_int_distribution<int> dist(0, std::get<5>(entry.second).size() - 1);
282  const PlantTypeConfig* pPTC = std::get<5>(entry.second).at(dist(EvoBeeModel::m_sRngEngine));
283  m_TargetReflectance.setVisDataPtr(pPTC->flowerVisDataPtr);
284  return;
285  }
286  }

290  std::stringstream msg;
291  msg << "Unexpected error encountered in the calculations in Hymenoptera::initialiseInnateTargetArbitrary()\n";
292  /*
293  msg << " Content of prefData map:\n";
294  float cumProbMax = 0.0;
295  for (auto& entry : prefData) {
296  msg << " " << entry.first << " -> (" << std::get<0>(entry.second) << ", " << std::get<1>(entry.second) << ", "
297  << std::get<2>(entry.second) << ", " << std::get<3>(entry.second) << ", "
298  << std::get<4>(entry.second) << ")\n";

303  msg << " selection==1 -> " << (EvoBee::equal(selection, 1.0f) ? "true" : "false") << "\n";
304  msg << " cumProbMax==1 -> " << (EvoBee::equal(cumProbMax, 1.0f) ? "true" : "false") << "\n";
305  msg << " cumProbMax>=selection -> " << ((cumProbMax>=selection) ? "true" : "false") << "\n";

566  }

577  }
578  }
579 
580  if (detectionProb > (1.0f - EvoBee::FLOAT_COMPARISON_EPSILON)) {

582  }
583  else if (detectionProb < (0.0f + EvoBee::FLOAT_COMPARISON_EPSILON)) {
584  return false;

592 // Determine whether the pollinator will decide to land on the given flower,
593 // using its visual perception. This entails three separate steps, (1) DETECT
594 // the flower, (2) DISTINGUISH whether or not the flower matches the pollinator's
595 // current target, and (3) DECIDE whether or not to land based upon the pollinator's
596 // learned probabilities of landing on a target or non-target flower as appropriate.
597 //
598 // This method performs the second and third of these steps (DISTINGUISH and DECIDE).
599 // The first step was already dealt with in Pollinator::forageNearestFlower.
600 //
601 // The second parameter, pJudgedToMatchTarget, is an optional pointer to a boolean
602 // variable used to return an indication of whether or not the pollinator judged
603 // the flower to match its target (whether or not it landed on it).
605 bool Hymenoptera::isVisitCandidateVisual(Flower* pFlower, bool* pJudgedToMatchTarget) const
606 {
607  bool bIsVisitCandidate = false;
608  bool bNoTargetSet = (getTargetWavelength() == NO_MARKER_POINT);
609  bool bJudgedToBeTarget = false;
610 
611  // (1) DETECT step
612  // We have already dealt with the pollinator's DETECT step by calling Pollinator::isDetected in
613  // Environment::findNearestUnvisitedFlower called from Pollinator::forageNearestFlower
614 
615  // (2) DISTINGUISH step
616  if (!bNoTargetSet)
617  {
618  if (m_sbVisTargetExactMatchOnly)
619  {
620  bJudgedToBeTarget = (getTargetWavelength() == pFlower->getCharacteristicWavelength());
621  }
622  else
623  {
624  float confidenceOfMatch = confidenceMatchesTarget(pFlower->getReflectanceInfo());
625  bJudgedToBeTarget = (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < confidenceOfMatch);
626  }
627  if (pJudgedToMatchTarget != nullptr) {
628  *pJudgedToMatchTarget = bJudgedToBeTarget;
629  }
630  }
631 
632  // (3) DECIDE step
633  // Here we make use of the pollinator's learned probabilities of landing on a target or non-target flower
634  // to make the final decision of whether to land
635  Wavelength flowerLambda = pFlower->getCharacteristicWavelength();
636  const VisualPreferenceInfo& visPrefInfo = getVisPrefInfoFromWavelengthConst(flowerLambda);
637 
638  if (bNoTargetSet)
639  {
640  float landProb = visPrefInfo.getProbLandNonTarget() + m_sVisProbLandNoTargetSetDelta;
641  bIsVisitCandidate = (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < landProb);
642  }
643  else
644  {
645  float landProb = bJudgedToBeTarget ? visPrefInfo.getProbLandTarget() : visPrefInfo.getProbLandNonTarget();
646  bIsVisitCandidate = (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < landProb);
647  }
648 
649  return bIsVisitCandidate;
650 }

685 // Update the pollinator's visual preference info according to the fickle-circumspect strategy
686 void Hymenoptera::updateVisualPrefsFickleCircumspect(const Flower* pFlower, int nectarCollected)
687 {
688  const ReflectanceInfo& flowerReflectance = pFlower->getReflectanceInfo();
689  Wavelength flowerLambda = flowerReflectance.getCharacteristicWavelength();
690  VisualPreferenceInfo& visPrefInfo = getVisPrefInfoFromWavelength(flowerLambda);
691  bool isTarget = (flowerLambda == getTargetWavelength());
692  bool noTarget = (getTargetWavelength() == NO_MARKER_POINT);
693  bool firstTarget = false;
694 
695  if (nectarCollected > 0)
696  {
697  // if the visitied flower was rewarding...
698  if (noTarget) {
699  setTargetWavelength(flowerReflectance.getVisDataPtr());
700  visPrefInfo.setAsTarget();
701  firstTarget = true;
702  noTarget = false;
703  }
704  else if (isTarget) {
705  visPrefInfo.incrementProbLandTarget(m_sVisProbLandIncrementOnReward);
706  }
707  else {
708  visPrefInfo.incrementProbLandNonTarget(m_sVisProbLandIncrementOnReward);
709  }
710  }
711  else
712  {
713  // if the visited flower was not rewarding...
714  if (isTarget) {
715  visPrefInfo.decrementProbLandTarget(m_sVisProbLandDecrementOnNoReward);
716  }
717  else {
718  visPrefInfo.decrementProbLandNonTarget(m_sVisProbLandDecrementOnNoReward);
719  }
720  }
721 
722  // consider switching to a new target flower based upon the newly updated preferences
723  if ((!firstTarget) && (!noTarget)) {
724  updateTarget();
725  }
726 
727  // attenuate preferences for non-target flowers that have not been recently encountered
728  attenuatePreferences();
729 }

746 // Consider switching to a new target flower colour based upon newly updated preferences
747 void Hymenoptera::updateTarget()
748 {
749  // this implementation only works with ColourSystem::REGULAR_MARKER_POINTS at this stage
750  // because the call to setTargetWavelength only takes a Wavelength at present - it really
751  // needs to take a ptr to a VisStimInfo object - so VisPrefInfo should really store one
752  // of those instead of a plain wavelength... TODO
753  assert(ModelParams::getColourSystem() == ColourSystem::REGULAR_MARKER_POINTS);
754 
755  float maxProbLandNonTarget = 0.0;
756  VisualPreferenceInfo& vpiCurrentTarget = getVisPrefInfoFromWavelength(getTargetWavelength());
757  VisualPreferenceInfo* pVpiMaxNonTarget = nullptr;
758 
759  // find the non-target marker point that currently has the highest landing probability
760  for (auto& vpi : m_VisualPreferences)
761  {
762  if ((vpi.getWavelength() != getTargetWavelength()) && (vpi.getProbLandNonTarget() > maxProbLandNonTarget))
763  {
764  pVpiMaxNonTarget = &vpi;
765  maxProbLandNonTarget = vpi.getProbLandNonTarget();
766  }
767  }
768 
769  // if the landing probability of the highest non-target marker point is higher than that of the
770  // current target marker point, make it the new target
771  if ((pVpiMaxNonTarget != nullptr) && (maxProbLandNonTarget > vpiCurrentTarget.getProbLandTarget()))
772  {
773  setTargetWavelength(pVpiMaxNonTarget->getWavelength());
774  // make the (old) non-target flower the new target
775 
776  pVpiMaxNonTarget->setAsTarget(); // set the (old) non-target flower's target prob land to the default target prob land
777 
778  vpiCurrentTarget.setAsNonTarget(); // set the (old) target flower's non-target prob land to the (old) target flower's target prob land
779  // (i.e. it used to have a high preference, keep it, but now use it as the non-target preference)
780  }
781 }
