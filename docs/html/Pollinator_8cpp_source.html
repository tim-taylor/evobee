225 /*
226  * Query whether this pollinator is currently within its allowed area,
227  * taking into account whether migration is allowed to outside of the
228  * Hive's initial area, and whether the new position is in a no-go
229  * area.
230  *
231  * NB This method also considers the migration restrictions defined for the
232  * Hive. If the pollinator has moved out of its previously allowed movement
233  * area and the migration is deemed to be allowed, then the pollinator's
234  * allowed movement area is updated according to the reproRestructionArea of
235  * the new patch. Hence, this is NOT a const method.
236  */
237 bool Pollinator::inAllowedArea()
238 {
239  bool migrated = false;
240 
241  bool ok = ((m_Position.x >= (float)m_MovementAreaTopLeft.x) &&
242  (m_Position.x < (float)(m_MovementAreaBottomRight.x+1)) &&
243  (m_Position.y >= (float)m_MovementAreaTopLeft.y) &&
244  (m_Position.y < (float)(m_MovementAreaBottomRight.y+1)));
245 
246  if ((!ok) && m_pHive->migrationAllowed())
247  {
248  if ((!m_pHive->migrationRestricted()) ||
249  (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < m_pHive->migrationProb()))
250  {
251  if (inEnvironment())
252  {
253  ok = true;
254  migrated = true;
255  }
256  }
257  }
258 
259  if (ok) {
260  Patch& patch = m_pEnv->getPatch(m_Position);
261  if (patch.noGoArea()) {
262  ok = false;
263  migrated = false;
264  }
265  }
266 
267  if (migrated) {
268  resetMovementArea();
269  }
270 
271  return ok;
272 }

317 // A helper method for the moveXXX methods for use when a pollinator has
318 // wandered out of the specified area. Reflect the movement off the edge of
319 // the environment and reposition the pollinator back within the allowed
320 // area
321 //
322 // @param delta is the change in position that the pollinator was trying to
323 // make from its current position, which would take it outside of its
324 // allowed area
325 void Pollinator::repositionInArea(fPos delta, float minx, float miny, float maxx, float maxy)
326 {
327  fPos oldPos = m_Position - delta;
328 
329  // to place the pollinator back in its allowed area, we'll need to reverse either
330  // the x or y component of its proposed move (or maybe both)
331  if (m_Position.x < minx || m_Position.x >= maxx)
332  {
333  delta.x = -delta.x;
334  }
335 
336  if (m_Position.y < miny || m_Position.y >= maxy)
337  {
338  delta.y = -delta.y;
339  }
340 
341  // now perform the movement with the adjusted delta
342  m_Position = oldPos + delta;
343 
344  // as a belt and braces measure, do one final check that the
345  // new position is within the allowed area
346  if (m_Position.x < minx)
347  {
348  m_Position.x = minx;
349  }
350  else if (m_Position.x >= maxx)
351  {
352  m_Position.x = maxx - (2.0 * EvoBee::FLOAT_COMPARISON_EPSILON);
353  }
354  if (m_Position.y < miny)
355  {
356  m_Position.y = miny;
357  }
358  else if (m_Position.y >= maxy)
359  {
360  m_Position.y = maxy - (2.0 * EvoBee::FLOAT_COMPARISON_EPSILON);
361  }
362 }

459 // The Random Global foraging strategy involves the pollinator picking a random flower
460 // across the entire environment. This effectively simulates a well-mixed population, i.e.
461 // one with no spatial aspects.
462 void Pollinator::forageRandomGlobal()
463 {
464  Flower* pFlower = nullptr;
465  bool flowerVisited = false;
466  unsigned int stepnum = m_pModel->getStepNumber();
467 
468  FlowerPtrVector& allFlowerPtrVec = getEnvironment()->getAllFlowerPtrVector();
469 
470  if (allFlowerPtrVec.size() > m_RecentlyVisitedFlowers.size())
471  {
472  do
473  {
474  // pick a random flower
475  std::uniform_int_distribution<unsigned int> dist(0, allFlowerPtrVec.size()-1);
476  pFlower = allFlowerPtrVec.at(dist(EvoBeeModel::m_sRngEngine));
477  unsigned int flowerID = pFlower->getId();
478 
479  // check whether it is on the recently visited list
480  if (std::find_if(m_RecentlyVisitedFlowers.begin(),
481  m_RecentlyVisitedFlowers.end(),
482  [flowerID](Flower* pOtherFlower){return (pOtherFlower->getId() == flowerID);})
483  != m_RecentlyVisitedFlowers.end())
484  {
485  // if so, reset pFlower to nullptr
486  pFlower = nullptr;
487  }
488 
489  // continue until we have found a flower that is not on the
490  // recently visited list
491  } while (pFlower == nullptr);
492 
493  if (isVisitCandidate(pFlower))
494  {
495  m_Position = pFlower->getPosition();
496  int rewardCollected = visitFlower(pFlower);
497  flowerVisited = true;
498  m_LatestAction.update(stepnum, PollinatorCurrentStatus::ON_FLOWER, pFlower, rewardCollected);
499  }
500  else
501  {
502  m_LatestAction.update(stepnum, PollinatorCurrentStatus::DECLINED_FLOWER, pFlower, 0);
503  }
504  }
505 
506  if (!flowerVisited)
507  {
508  // there's no point in doing a random move if the flower is not visited,
509  // because next time we'll be chosing a new flower at random from the whole
510  // population in any case
511 
512  losePollenToAir(m_iPollenLossInAir);
513  }

610 // Determine whether the pollinator will decide to land on the given flower,
611 // which depends on its flower constancy characteristics etc.
612 //
613 // The second parameter, pJudgedToMatchTarget, is an optional pointer to a boolean
614 // variable used to return an indication of whether or not the pollinator judged
615 // the flower to match its target (whether or not it landed on it). This is only
616 // relevant for visual foraging (foraging-strategy='nearest-flower' and
617 // constancy-type='visual').
618 //
619 bool Pollinator::isVisitCandidate(Flower* pFlower, bool* pJudgedToMatchTarget) const
620 {
621  bool bIsVisitCandidate = true;
622 
623  switch (m_ConstancyType)
624  {
625  case (PollinatorConstancyType::NONE):
626  {
627  bIsVisitCandidate = true;
628  break;
629  }
630  case (PollinatorConstancyType::SIMPLE):
631  {
632  // take into account the type of the previous flower visited, and the probability
633  // of landing on the same type this time around...
634 
635  if (m_PreviousLandingSpeciesId == 0)
636  {
637  // have not landed on a flower previously, so we'll land on this one
638  // whatever!
639  bIsVisitCandidate = true;
640  }
641  else if (m_PreviousLandingSpeciesId == pFlower->getSpeciesId())
642  {
643  // land on the same species of flower with a high fixed prob
644  float prob = 0.9;
645  bIsVisitCandidate = (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < prob);
646  }
647  else {
648  // land on a different species of flower with prob determined by constancy param
649  float prob = 1.0 - m_fConstancyParam;
650  bIsVisitCandidate = (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < prob);
651  }
652 
653  break;
654  }
655  case (PollinatorConstancyType::VISUAL):
656  {
657  bIsVisitCandidate = isVisitCandidateVisual(pFlower, pJudgedToMatchTarget);
658  break;

702 // The logic of what a pollinator does when it lands on a flower.
703 // This method implement transfer of pollen between the pollinator and the flower
704 // (in both directions), collection of nectar by the pollinator, updating of the
705 // pollinator's visual preferences according to its learning strategy (if relevant),
706 // and various bits of housekeeping.
707 //
708 // Returns the amount of nectar collected on this visit.
709 //
710 int Pollinator::visitFlower(Flower* pFlower)
711 {
712  // for each Pollen grain in the store, update its landing count
713  updatePollenLandingCount();
714 
715  // remove pollen from store that is now past the carryover limit
716  removeOldCarryoverPollen();
717 
718  // first transfer some of bee's pollen to the flower (potentially pollinating it)
719  depositPollenOnStigma(pFlower);
720 
721  // now pick up more pollen from the flower
722  collectPollenFromAnther(pFlower);
723 
724  // and collect nectar reward if available
725  int nectarCollected = collectNectar(pFlower);
726 
727  // if pollinator is using visual system for foraging, update its preferences after visiting this flower
728  if (m_ConstancyType == PollinatorConstancyType::VISUAL)
729  {
730  updateVisualPreferences(pFlower, nectarCollected);
731  }
732 
733  // update record of most recent landing to this one
734  m_PreviousLandingSpeciesId = pFlower->getSpeciesId();
735  if (m_uiVisitedFlowerMemorySize == 1)
736  {
737  if (m_RecentlyVisitedFlowers.empty())
738  {
739  m_RecentlyVisitedFlowers.push_back(pFlower);
740  }
741  else
742  {
743  m_RecentlyVisitedFlowers[0] = pFlower;
744  }
745  }
746  else if (m_uiVisitedFlowerMemorySize > 1)
747  {
748  m_RecentlyVisitedFlowers.push_back(pFlower);
749  if (m_RecentlyVisitedFlowers.size() > m_uiVisitedFlowerMemorySize)
750  {
751  m_RecentlyVisitedFlowers.erase(m_RecentlyVisitedFlowers.begin());
752  }
753  }
754 
755  // update count of number of flowers visited, and end bout if done
756  ++m_iNumFlowersVisitedInBout;
757  if ((m_iBoutLength > 0) && (m_iNumFlowersVisitedInBout >= m_iBoutLength))
758  {
759  m_State = PollinatorState::BOUT_COMPLETE;
760  }
761 
762  return nectarCollected;
763 }

864 // Collect pollen from flower if available, and add to our store
865 // If any pollen is added, the contents of the store is shuffled so that
866 // subsequent operations that act of randomly chosen grains from the store
867 // can simply and efficiently pick off grains from the end of the store.
868 // If the store's maxium capacity is exceeded, some random grains are
869 // removed to bring it down to the maximum allowed size.
870 //
871 void Pollinator::collectPollenFromAnther(Flower* pFlower)
872 {
873  // transfer anther pollen from flower to our store
874  int num = pFlower->transferAntherPollenToPollinator(m_PollenStore);
875 
876  // if some new grains have been added, shuffle the whole store now
877  // so that subsequent removals of pollen happen in a random order
878  if (num > 0)
879  {
880  std::shuffle(m_PollenStore.begin(), m_PollenStore.end(), EvoBeeModel::m_sRngEngine);
881  }
882 
883  // finally, if the number of grains in our store now exceeds the
884  // maximum capacity, delete some. As we've just shuffled the store,
885  // we can just remove the required number from the end of the store
886  int xs = m_PollenStore.size() - m_iMaxPollenCapacity;
887  if (xs > 0)
888  {
889  m_PollenStore.erase(m_PollenStore.end()-xs, m_PollenStore.end());
890  }
891 }

966 // This method represents the pollinator's ability to discriminate between a given stimulus and
967 // the pollinator's remembered target. The method returns a confidence level (between 0.0 and 1.0)
968 // that the given stimulus *matches* the current target (i.e. the probability that the pollinator
969 // does *not* descriminate a difference between the two).
970 //
971 float Pollinator::confidenceMatchesTarget(const ReflectanceInfo& stimulus) const
972 {
973  const float minHexDistance = 0.05;
974  const float maxHexDistance = 0.19;
975  const float maxConfidence = 0.95;
976  const float minConfidence = 0.05;
977 
978  float confidence = minConfidence;
979 
980  float hexDistance;
981 
982  switch (ModelParams::getColourSystem())
983  {
984  case ColourSystem::REGULAR_MARKER_POINTS:
985  {
986  const VisualStimulusInfo& infoStimulus = getVisStimulusInfo(stimulus.getMarkerPoint());
987  const VisualStimulusInfo& infoTarget = getVisStimulusInfo(getTargetWavelength());
988  hexDistance = getVisHexDistance(infoStimulus, infoTarget, false); // calculate distance using raw positions in hex space
989  break;
990  }
991  case ColourSystem::ARBITRARY_DOMINANT_WAVELENGTHS:
992  {
993  const VisualStimulusInfo* pInfoStimulus = stimulus.getVisDataPtr();
994  const VisualStimulusInfo* pInfoTarget = m_TargetReflectance.getVisDataPtr();
995  hexDistance = getVisHexDistance(*pInfoStimulus, *pInfoTarget, false); // calculate distance using raw positions in hex space
996  break;
997  }
998  default:
999  {
1000  throw std::runtime_error("Encountered an unknown ColourSystem in Pollinator::confidenceMatchesTarget()");
1001  }
1002  }
1003 
1004  if (hexDistance <= minHexDistance)
1005  {
1006  confidence = maxConfidence;
1007  }
1008  else if (hexDistance >= maxHexDistance)
1009  {
1010  confidence = minConfidence;
1011  }
1012  else
1013  {
1014  float distanceFrac = (hexDistance - minHexDistance) / (maxHexDistance - minHexDistance);
1015  confidence = maxConfidence - distanceFrac * (maxConfidence - minConfidence);
1016  }
1017 
1018  return confidence;
1019 }
